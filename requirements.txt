# latent_text_gen — dependencies (Linux GPU server)
#
# INSTALL ORDER MATTERS. Do NOT independently pin torch/fairseq2/CUDA here:
# that is exactly how the fairseq2 <-> torch <-> CUDA conflicts bite. Instead:
#
#   1. Create a fresh env (python 3.10/3.11):
#        conda create -n ltg python=3.11 -y && conda activate ltg
#   2. Install SONAR FIRST and let it resolve torch/fairseq2 for your CUDA:
#        pip install sonar-space
#   3. Install the remaining (codec-agnostic) deps:
#        pip install -r requirements.txt
#   4. Freeze the *resolved* set for reproducibility on this box:
#        pip freeze > requirements.lock.txt
#
# The packages below deliberately avoid torch/fairseq2/sonar-space version pins.
